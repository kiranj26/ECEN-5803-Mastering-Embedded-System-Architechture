51static inline void gpio_write(gpio_t *obj, int value) {
52 MBED_ASSERT(obj->pin != (PinName)NC);
53 if (value) {
54 *obj->reg_set = obj->mask;
55 } else {
56 *obj->reg_clr = obj->mask;
57 }
58}

60static inline int gpio_read(gpio_t *obj) {
61 MBED_ASSERT(obj->pin != (PinName)NC);
62 return ((*obj->reg_in & obj->mask) ? 1 : 0);
63}
